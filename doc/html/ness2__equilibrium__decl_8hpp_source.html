155 template <typename DOS>
156 void green_equilibrium_ness_fermi(herm_matrix_ness &G, DOS &dos, double beta,double mu,
157 double h,fft_integral_method method, int limit, int nn) {
158 if(method==FFT_ADAPTIVE){
159 // Retarded Green’s function
160 green_equilibrium_ret_ness_adaptive(G,dos,h,nn,limit);
161 // Lesser Green’s function
162 auto fermi_integrand = [&](double omega) {
163 double arg = beta * (omega - mu);
164 double fm;
165 if (arg > 100) fm=0.0;
166 else if (arg < -100) fm=1.0;
167 else fm=1.0/(1.0 + std::exp(arg));
168 return -dos(omega) *fm;
169 };
170 // Compute lesser Green’s function
171 compute_lesser(G.les_,G.Nft_/2,fermi_integrand,dos.lo_,dos.hi_,h,nn,limit);
172 }else{
173 // METHOD=TRAPEZ
174 int Nft=G.Nft_;
175 int size1=G.size1_;
176 grid_info grid(Nft,h);
177 cdmatrix Id = cdmatrix::Identity(size1, size1);
178 int nfreq=Nft/2;
179 G.ret_.set_zero(fft_domain::freq);
180 for(int w = -nfreq+1; w < nfreq; ++w){
181 double omega=grid.freq_at(w);
182 double aw=dos(omega);
183 double fm;
184 cdmatrix tmp=Id*cdouble(0,-M_PI*aw);
185 // set only Im(Gret), as only Re(Gret) is not used in backtransform
186 G.set_ret(w,tmp,fft_domain::freq);
187 double arg = beta * (omega - mu);
188 if (arg > 100) fm=0.0;
189 else if (arg < -100) fm=1.0;
190 else fm=1.0/(1.0 + std::exp(arg));
191 G.set_les(w,-2*fm*tmp,fft_domain::freq);
192 }
193 // this transforms back the imaginary part of G(w) only
194 G.integral_transform_to_time(h,FFT_TRAPEZ);
195 }
196 return;
197 }

200 template <typename DOS>
201 void green_equilibrium_ness_bose(herm_matrix_ness &G, DOS &dos, double beta, double mu,
202 double h,fft_integral_method method, int limit,int nn) {
203 auto bose_integrand = [&](double omega) {
204 // A(w)*bose(w-mu), assuming that A(w)=0 at w=mu
205 double arg = beta * (omega - mu);
206 if (arg > 100) return 0.0;
207 if (arg < -100) return -dos(omega);
208 if (std::abs(arg) < 1e-8) {
209 // Use small-argument limit: A(w) ~ (w-mu) assumed:
210 double delta = 1e-6;
211 double slope = (dos(delta + mu) - dos(mu)) / delta;
212 return slope / beta;
213 }
214 return dos(omega) / (std::exp(arg) - 1.0);
215 };
216 if(method==FFT_ADAPTIVE){
217 green_equilibrium_ret_ness_adaptive(G,dos,h,nn,limit);
218 compute_lesser(G.les_,G.Nft_/2, bose_integrand,dos.lo_,dos.hi_, h, nn, limit);
219 }else{
220 //method==FFT_TRAPEZ
221 int Nft=G.Nft_;
222 int size1=G.size1_;
223 grid_info grid(Nft,h);
224 cdmatrix Id = cdmatrix::Identity(size1, size1);
225 int nfreq=Nft/2;
226 G.ret_.set_zero(fft_domain::freq);
227 for(int w = -nfreq+1; w < nfreq; ++w){
228 double omega=grid.freq_at(w);
229 double aw=dos(omega);
230 cdouble fm=cdouble(0,-bose_integrand(omega)*2*M_PI);
231 cdmatrix tmp=Id*cdouble(0,-M_PI*aw);
232 G.set_ret(w,tmp,fft_domain::freq);
233 tmp=Id*fm;
234 G.set_les(w,tmp,fft_domain::freq);
235 }
236 // this transforms back the imaginary part of G(w) only
237 G.integral_transform_to_time(h,FFT_TRAPEZ);
238 }
239 }

254 template <typename DosFunction>
255 void green_equilibrium_ret_ness_adaptive(herm_matrix_ness &G, DosFunction &dos, double h, int nn,int limit) {
256 int nt = G.Nft_/2;
257 int size1 = G.retarded().size1_;
258 int Ntot = 2 * nt;
259 fourier::adft_func adft;
260 auto integrand = [&](double omega) {
261 return dos(omega);
262 };
263 adft.sample(0.0, dos.lo_, dos.hi_, integrand, nn, limit);
264 // Compute positive times t = n h
265 for (int n = 0; n < nt; ++n) {
266 cplx res, err;
267 double tval = h * n;
268 adft.dft(-tval, res, err);
269 res *= cplx(0, -1.0); // apply prefactor -i
270 Eigen::MatrixXcd M = Eigen::MatrixXcd::Zero(size1, size1);
271 for (int i = 0; i < size1; ++i) {
272 M(i, i) = res;
273 }
274 G.retarded().set_element(n, M, fft_domain::time);
275 }
276 // Set negatice times to 0:
277 Eigen::MatrixXcd zeroMat = Eigen::MatrixXcd::Zero(size1, size1);
278 for (int n = nt; n < 2*nt; ++n) {
279 G.retarded().set_element(n, zeroMat, fft_domain::time);
280 }
281 }

284 template <typename Integrand>
285 void compute_lesser(fft_array &G, int nt, Integrand &integrand, double lo, double hi, double h, int nn, int limit) {
286 int Nft = G.Nft_;
287 int size = G.size1_;
288 assert(size > 0);
289 assert(nt <= Nft/2);
290
291 G.set_zero(fft_domain::time);
292 fourier::adft_func adft;
293 adft.sample(10*h, lo, hi, integrand, nn, limit);
294
295 // Prepare the full matrix at each time
296 for (int n = 0; n < nt; ++n) {
297 double t = n * h;
298 std::complex<double> res, err;
299 adft.dft(-t, res, err);
300 res *= std::complex<double>(0, -1.0);
301
302 Eigen::MatrixXcd mat(size, size);
303 mat.setZero();
304 mat.diagonal().setConstant(res);
305
306 G.set_element(n, mat, fft_domain::time);
307
308 if (n > 0) {
309 Eigen::MatrixXcd mat_neg = -mat.adjoint();
310 G.set_element(-n, mat_neg, fft_domain::time);
311 }
312 }
313
314 // Set midpoint (Nft / 2) to zero matrix
315 Eigen::MatrixXcd zero_mat = Eigen::MatrixXcd::Zero(size, size);
316 G.set_element(Nft / 2, zero_mat, fft_domain::time);
317 }
