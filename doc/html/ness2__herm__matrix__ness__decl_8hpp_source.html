436 template<typename T>
437 void herm_matrix_ness::read_from_cntr(cntr::herm_matrix<T> &Gcntr,int tstp){
438 if(size1_!=Gcntr.size1() || tstp> Gcntr.nt()){
439 throw std::runtime_error("dyson_ret read_ness_from_cntr:input param mismatch");
440 }
441 cdmatrix M;
442 this->set_zero(fft_domain::time);
443 for(int i=0;i<Nft_/2;i++){
444 if(i<=tstp){
445 Gcntr.get_ret(tstp,tstp-i,M);
446 this->set_ret(i,M,fft_domain::time);
447 Gcntr.get_les(tstp,tstp-i,M);
448 this->set_les(i,M,fft_domain::time);
449 if(i>0) this->set_les(-i,-M.adjoint(),fft_domain::time);
450 }
451 }
452 }
