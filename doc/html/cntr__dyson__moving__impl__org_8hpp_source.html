110 element_linsolve_right<T,SIZE1>(size1,G.retptr(0,j),mm,qq);
111 }
112 delete [] stemp;
113 delete [] qq;
114 delete [] mm;
115 delete [] gtemp;
116 delete [] one;
117 delete [] diffw;
