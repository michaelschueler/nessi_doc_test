160 void cntr2ness(herm_matrix_ness& Gness, cntr::herm_matrix<T>& Gcntr, int tstp = -1) {
161 int Nft_half = Gness.Nft_ / 2;
162 int size = Gness.size1_;
163 int nt=Gcntr.nt();
164 if (size != Gcntr.size1()) {
165 throw std::runtime_error("cntr2ness: matrix size mismatch");
166 }
167 if (tstp>nt || nt<0) {
168 throw std::runtime_error("cntr2ness: wrong dimensiuons in Gcntr");
169 }
170 if (tstp < 0) tstp = Gcntr.nt(); // default to max physical time
171 Gness.set_zero(fft_domain::time);
172 cdmatrix M;
173 for (int t = 0; t <= std::min(tstp, Nft_half - 1); ++t) {
174 Gcntr.get_ret(tstp, tstp - t, M);
175 Gness.set_ret(t, M, fft_domain::time);
176 }
177 for (int t = -std::min(tstp, Nft_half - 1); t <= 0; ++t) {
178 Gcntr.get_les(tstp+t, tstp, M);
179 Gness.set_les(t, M, fft_domain::time);
180 if (t < 0) Gness.set_les(-t, -M.adjoint(), fft_domain::time);
181 }
